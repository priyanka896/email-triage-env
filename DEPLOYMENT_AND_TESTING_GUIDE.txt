import sys
sys.path.insert(0, ".")

from email_triage_env.models import Priority, Category

def test_easy_task():
    env = EmailTriageEnvironment()
    obs = env.reset(task_id="easy_triage")

    assert obs.task_id == "easy_triage"
    assert obs.email_subject == "Cannot log in to my account"
    assert obs.emails_remaining == 3
    assert obs.done is False
    print("[PASS] Easy task reset")

    # Email 1: Login issue → urgent, account_access
    obs = env.step(EmailAction(
        priority=Priority.URGENT,
        category=Category.ACCOUNT_ACCESS,
        reply="I understand you cannot log in. Let me help reset your password and restore account access.",
        escalate=False,
    ))
    assert obs.reward == 1.0, f"Expected 1.0, got {obs.reward}"
    print(f"[PASS] Easy email 1: reward={obs.reward}")

    # Email 2: Dark mode → low, feature_request
    obs = env.step(EmailAction(
        priority=Priority.LOW,
        category=Category.FEATURE_REQUEST,
        reply="Thanks for the feature request! We will add dark mode to our roadmap.",
        escalate=False,
    ))
    assert obs.reward == 1.0, f"Expected 1.0, got {obs.reward}"
    print(f"[PASS] Easy email 2: reward={obs.reward}")

    # Email 3: Billing charge → high, billing
    obs = env.step(EmailAction(
        priority=Priority.HIGH,
        category=Category.BILLING,
        reply="I see the charge on your account. Let me look into this billing issue.",
        escalate=False,
    ))
    assert obs.done is True
    assert obs.reward == 1.0, f"Expected avg 1.0, got {obs.reward}"
    print(f"[PASS] Easy email 3 (final): avg_reward={obs.reward}")
    print(f"[PASS] Easy task complete. State: steps={env.state.step_count}")

def test_partial_scoring():
    """Verify partial credit works correctly."""
    env = EmailTriageEnvironment()
    env.reset(task_id="easy_triage")

    # Completely wrong answer
    obs = env.step(EmailAction(
        priority=Priority.LOW,
        category=Category.SPAM,
        reply="ok",
        escalate=True,
    ))
    assert obs.reward < 0.2, f"Wrong answer should score low, got {obs.reward}"
    print(f"[PASS] Partial scoring — wrong answer: reward={obs.reward}")

    # Adjacent priority (medium instead of low)
    obs = env.step(EmailAction(
        priority=Priority.MEDIUM,
        category=Category.FEATURE_REQUEST,
        reply="Thanks for the feature request!",
        escalate=False,
    ))
    assert 0.5 < obs.reward < 1.0, f"Adjacent should get partial, got {obs.reward}"
    print(f"[PASS] Partial scoring — adjacent priority: reward={obs.reward}")


def test_state():
    """Verify state tracking."""
    env = EmailTriageEnvironment()
    env.reset(task_id="easy_triage")

    assert env.state.step_count == 0
    assert env.state.current_email_index == 0
    assert env.state.total_emails == 3

    env.step(EmailAction(
        priority=Priority.URGENT,
        category=Category.ACCOUNT_ACCESS,
        reply="Resetting your password and account access now.",
        escalate=False,
    ))

    assert env.state.step_count == 1
    assert env.state.current_email_index == 1
    assert env.state.cumulative_reward > 0
    print(f"[PASS] State tracking: steps={env.state.step_count}, cumulative={env.state.cumulative_reward}")


def test_reset_clears_state():
    """Verify reset clears previous episode."""
    env = EmailTriageEnvironment()
    env.reset(task_id="easy_triage")
    env.step(EmailAction(
        priority=Priority.URGENT,
        category=Category.ACCOUNT_ACCESS,
        reply="Resetting your password now.",
        escalate=False,
    ))
    assert env.state.step_count == 1

    # Reset should clear everything
    obs = env.reset(task_id="medium_triage")
    assert env.state.step_count == 0
    assert env.state.current_email_index == 0
    assert obs.task_id == "medium_triage"
    print("[PASS] Reset clears state correctly")


def test_invalid_task_fallback():
    """Verify invalid task_id falls back to easy."""
    env = EmailTriageEnvironment()
    obs = env.reset(task_id="nonexistent_task")
    assert obs.task_id == "easy_triage"

import asyncio
import json
import websockets

async def test_websocket():
    for task_id in ["easy_triage", "medium_triage", "hard_triage"]:
        async with websockets.connect("ws://localhost:8000/ws") as ws:
            # Reset
            await ws.send(json.dumps({"type": "reset", "data": {"task_id": task_id}}))
            resp = json.loads(await ws.recv())
            envelope = resp["data"]
            obs = envelope["observation"]
            done = envelope.get("done", False)

            print(f"\nTask: {task_id}")
            print(f"  First email: {obs['email_subject']}")

            step = 0
            while not done:
                action = {
                    "priority": "medium",
                    "category": "general_inquiry",
                    "reply": "Thank you for contacting us. We are looking into this.",
                    "escalate": False,
                }
                await ws.send(json.dumps({"type": "step", "data": action}))
                resp = json.loads(await ws.recv())
                envelope = resp["data"]
                obs = envelope["observation"]
                reward = envelope.get("reward", 0)
                done = envelope.get("done", False)
                step += 1
                print(f"  Step {step}: reward={reward:.4f}, done={done}")

            print(f"  Final score: {envelope['reward']:.4f}")

asyncio.run(test_websocket())
--- END test_ws.py ---

================================================================================
PART 4: TROUBLESHOOTING
================================================================================

Problem: Docker build fails with "openenv-core not found"
  → Make sure pyproject.toml has: openenv-core>=0.2.2 in dependencies

Problem: WebSocket connection refused on HF Space
  → HF Spaces may need the URL format: wss://YOUR_USERNAME-email-triage-env.hf.space/ws
  → Check Space logs for startup errors

Problem: inference.py times out
  → Increase MAX_STEPS if needed (default 10)
  → Check that ENV_URL points to the correct running server
  → Verify your HF_TOKEN has API access to the model

Problem: openenv validate fails
  → Ensure openenv.yaml is in the email_triage_env/ directory
  → Ensure server is running when you run validate

Problem: Scores are all 0.0
  → Check that the LLM is returning valid JSON
  → Set DEBUG=True in inference.py to see raw LLM responses
  → Verify the model name is correct and accessible
